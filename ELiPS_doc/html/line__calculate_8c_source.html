352  static fp2_t tmp1_fp2, tmp2_fp2, tmp3_fp2, tmp4_fp2, tmp5_fp2;
353  static fp12_t tmp1_fp12;
354  efp2_t Tmp_T;
355  efp2_init(&Tmp_T);
356  efp2_set(&Tmp_T, T);
357 
358  // ltq
359  fp2_sub_mod(&tmp1_fp2, &Q->x, &Tmp_T.x); // tmp1_fp2=(Q->x-T.x)^-1
360  fp2_inv_montgomery(&tmp1_fp2, &tmp1_fp2);
361  fp2_sub_nonmod(&tmp2_fp2, &Q->y, &Tmp_T.y); // tmp2_fp2=(Q->y-T.y)
362  fp2_mul_mod_montgomery(&tmp3_fp2, &tmp1_fp2, &tmp2_fp2); // tmp3_fp2=tmp1_fp2*tmp2_fp2
363  fp2_add_mod(&tmp4_fp2, &Tmp_T.x, &Q->x); // tmp4_fp2=Q->x+T.x
364  fp2_sqr_mod_montgomery(&T->x, &tmp3_fp2); // next_T.x=tmp3_fp2^2-tmp4_fp2
365  fp2_sub_mod(&T->x, &T->x, &tmp4_fp2);
366  fp2_mul_mod_montgomery(&tmp5_fp2, &tmp3_fp2, &Tmp_T.x); // tmp5_fp2=tmp3_fp2*T.x-T.y
367  fp2_sub_mod(&tmp5_fp2, &tmp5_fp2, &Tmp_T.y);
368  fp2_mul_mod_montgomery(&T->y, &tmp3_fp2, &T->x); // next_T.y=tmp5_fp2-tmp3_fp2*next_T.x
369  fp2_sub_mod(&T->y, &tmp5_fp2, &T->y);
370  // set ltq
371 #ifdef TWIST_PHI_INV
372  fp_set_mpn(&tmp1_fp12.x0.x0.x0, RmodP);
373  fp2_set_neg(&tmp1_fp12.x1.x0, &tmp3_fp2);
374  fp2_mul_fp_mod_montgomery(&tmp1_fp12.x1.x1, &tmp5_fp2, L);
375  fp12_8sparse_mul_mod_montgomery(f, f, &tmp1_fp12);
376 #endif
377 #ifdef TWIST_PHI
378  fp_set_mpn(&tmp1_fp12.x1.x1.x0, RmodP);
379  fp2_set_neg(&tmp1_fp12.x0.x1, &tmp3_fp2);
380  fp2_mul_fp_mod_montgomery(&tmp1_fp12.x0.x0, &tmp5_fp2, L);
381  fp12_8sparse_mul_mod_montgomery(f, f, &tmp1_fp12);
382 #endif
383 }

419  fp2_mul_mod_montgomery(&T->z, &T->z, &tmp3_fp2);

432  fp_mul_mod_montgomery(&tmp1_fp12.x1.x1.x1, &tmp1_fp2.x1, &P->y);
433 #endif
434  fp12_6sparse_mul_mod_montgomery(f, f, &tmp1_fp12);
