259  fp_init(&tmp0);

262  if (fp2_cmp_zero(A) == 0) {
263  fp2_set(ANS, A);

267  fp_set_neg_mod(&tmp0, &A->x0);

269  if (ret == 1) {

271  fp_set(&ANS->x1, &tmp0);
272  return 1;
273  }
274

276  if (ret == 1) {
277  fp_set(&ANS->x0, &tmp0);
278  fp_set_ui(&ANS->x1, 0);
279  return 1;
280  }
281  return 0;

284  fp2_set(&tmp_A, A);
285  fp_sqr_mod_montgomery(&tmp0, &tmp_A.x0);
286  fp_sqr_mod_montgomery(&tmp1, &tmp_A.x1);
287  fp_add_mod(&tmp0, &tmp0, &tmp1); // tmp0 = a0^2 - (a1 * alpha)^2
288  ret = fp_legendre_sqrt_montgomery(&tmp0, &tmp0);
289  if (ret != 1) {
290  return 0;
291  }
292  fp_add_mod(&tmp1, &tmp_A.x0, &tmp0);
293  fp_r1shift(&tmp1, &tmp1);

295  if (ret != 1) {

297  fp_r1shift(&tmp1, &tmp1);
